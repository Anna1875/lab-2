<script>
// –§—É–Ω–∫—Ü—ñ–æ–Ω–∞–ª—å–Ω—ñ—Å—Ç—å –ø–æ—à—É–∫—É —Ç–∞ —Ñ—ñ–ª—å—Ç—Ä–∞—Ü—ñ—ó –≥–ª–æ—Å–∞—Ä—ñ—é
document.addEventListener('DOMContentLoaded', function() {
  const searchInput = document.getElementById('search-terms');
  const filterButtons = document.querySelectorAll('.filter-btn');
  const glossaryTerms = document.querySelectorAll('.glossary-term');
  
  let currentFilter = 'all';
  
  // –ü–æ—à—É–∫ –ø–æ —Ç–µ—Ä–º—ñ–Ω–∞—Ö
  searchInput.addEventListener('input', function() {
    const searchTerm = this.value.toLowerCase();
    filterTerms(searchTerm, currentFilter);
  });
  
  // –§—ñ–ª—å—Ç—Ä–∞—Ü—ñ—è –ø–æ –∫–∞—Ç–µ–≥–æ—Ä—ñ—è—Ö
  filterButtons.forEach(button => {
    button.addEventListener('click', function() {
      // –û–Ω–æ–≤–ª—é—î–º–æ –∞–∫—Ç–∏–≤–Ω—É –∫–Ω–æ–ø–∫—É
      filterButtons.forEach(btn => btn.classList.remove('btn-primary'));
      filterButtons.forEach(btn => btn.classList.add('btn-outline'));
      this.classList.remove('btn-outline');
      this.classList.add('btn-primary');
      
      currentFilter = this.dataset.filter;
      const searchTerm = searchInput.value.toLowerCase();
      filterTerms(searchTerm, currentFilter);
    });
  });
  
  // –§—É–Ω–∫—Ü—ñ—è —Ñ—ñ–ª—å—Ç—Ä–∞—Ü—ñ—ó
  function filterTerms(searchTerm, category) {
    glossaryTerms.forEach(term => {
      const termText = term.textContent.toLowerCase();
      const termCategory = term.dataset.category;
      const termKey = term.dataset.term;
      
      const matchesSearch = !searchTerm || 
                           termText.includes(searchTerm) || 
                           (termKey && termKey.includes(searchTerm));
      
      const matchesCategory = category === 'all' || termCategory === category;
      
      if (matchesSearch && matchesCategory) {
        term.style.display = 'block';
        
        // –ü—ñ–¥—Å–≤—ñ—á—É—î–º–æ –∑–Ω–∞–π–¥–µ–Ω–∏–π —Ç–µ–∫—Å—Ç
        if (searchTerm && matchesSearch) {
          highlightSearchTerm(term, searchTerm);
        } else {
          removeHighlight(term);
        }
      } else {
        term.style.display = 'none';
      }
    });
  }
  
  // –ü—ñ–¥—Å–≤—ñ—á—É–≤–∞–Ω–Ω—è –∑–Ω–∞–π–¥–µ–Ω–æ–≥–æ —Ç–µ–∫—Å—Ç—É
  function highlightSearchTerm(element, searchTerm) {
    removeHighlight(element);
    
    const walker = document.createTreeWalker(
      element,
      NodeFilter.SHOW_TEXT,
      null,
      false
    );
    
    const textNodes = [];
    let node;
    while (node = walker.nextNode()) {
      textNodes.push(node);
    }
    
    textNodes.forEach(textNode => {
      const text = textNode.textContent;
      const regex = new RegExp(`(${searchTerm})`, 'gi');
      if (regex.test(text)) {
        const highlightedHTML = text.replace(regex, '<mark style="background: yellow; padding: 0;">$1</mark>');
        const span = document.createElement('span');
        span.innerHTML = highlightedHTML;
        textNode.parentNode.replaceChild(span, textNode);
      }
    });
  }
  
  // –í–∏–¥–∞–ª–µ–Ω–Ω—è –ø—ñ–¥—Å–≤—ñ—á—É–≤–∞–Ω–Ω—è
  function removeHighlight(element) {
    const marks = element.querySelectorAll('mark');
    marks.forEach(mark => {
      mark.outerHTML = mark.innerHTML;
    });
    
    const spans = element.querySelectorAll('span');
    spans.forEach(span => {
      if (span.innerHTML.includes('mark')) {
        span.outerHTML = span.innerHTML;
      }
    });
  }
  
  // –ü–æ—á–∞—Ç–∫–æ–≤–µ –≤—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–Ω—è –∞–∫—Ç–∏–≤–Ω–æ—ó –∫–Ω–æ–ø–∫–∏
  filterButtons[0].classList.remove('btn-outline');
  filterButtons[0].classList.add('btn-primary');
});
