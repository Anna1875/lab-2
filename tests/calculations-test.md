      const incorrectOption = questionDiv.querySelector(`input[value="${userAnswer}"]`).closest('.answer-option');
      incorrectOption.classList.add('incorrect');
    }
    
    // –ü–æ–∑–Ω–∞—á–∞—î–º–æ –≤–∏–±—Ä–∞–Ω—É –≤—ñ–¥–ø–æ–≤—ñ–¥—å
    if (userAnswer) {
      const selectedOption = questionDiv.querySelector(`input[value="${userAnswer}"]`).closest('.answer-option');
      selectedOption.classList.add('selected');
    }
  }
  
  // –ü–æ–∫–∞–∑—É—î–º–æ —Ä–µ–∑—É–ª—å—Ç–∞—Ç
  showResult(correctCount, totalQuestions);
}

// –§—É–Ω–∫—Ü—ñ—è –ø–æ–∫–∞–∑—É —Ä–µ–∑—É–ª—å—Ç–∞—Ç—É
function showResult(correct, total) {
  const percentage = Math.round((correct / total) * 100);
  let resultClass = 'error';
  let resultText = '–ü–æ—Ç—Ä—ñ–±–Ω–æ –∫—Ä–∞—â–µ –≤–∏–≤—á–∏—Ç–∏ –º–∞—Ç–µ–º–∞—Ç–∏—á–Ω—ñ –æ—Å–Ω–æ–≤–∏';
  let grade = 'F';
  
  if (percentage >= 90) {
    resultClass = 'success';
    resultText = '–í—ñ–¥–º—ñ–Ω–Ω—ñ –º–∞—Ç–µ–º–∞—Ç–∏—á–Ω—ñ –Ω–∞–≤–∏—á–∫–∏!';
    grade = 'A';
  } else if (percentage >= 80) {
    resultClass = 'success';
    resultText = '–î–æ–±—Ä–µ –≤–æ–ª–æ–¥—ñ—î—Ç–µ —Ä–æ–∑—Ä–∞—Ö—É–Ω–∫–∞–º–∏!';
    grade = 'B';
  } else if (percentage >= 70) {
    resultClass = 'warning';
    resultText = '–ù–µ–ø–æ–≥–∞–Ω–æ, –∞–ª–µ –ø–æ—Ç—Ä—ñ–±–Ω–∞ –ø—Ä–∞–∫—Ç–∏–∫–∞';
    grade = 'C';
  } else if (percentage >= 60) {
    resultClass = 'warning';
    resultText = '–ë–∞–∑–æ–≤—ñ –Ω–∞–≤–∏—á–∫–∏ —î';
    grade = 'D';
  } else if (percentage >= 50) {
    resultClass = 'warning';
    resultText = '–ú—ñ–Ω—ñ–º–∞–ª—å–Ω–∏–π —Ä—ñ–≤–µ–Ω—å';
    grade = 'E';
  }
  
  const resultDiv = document.getElementById('quiz-result');
  resultDiv.innerHTML = `
    <div class="test-result ${resultClass}">
      <h3>${resultText}</h3>
      <p><strong>–ü—Ä–∞–≤–∏–ª—å–Ω–∏—Ö –≤—ñ–¥–ø–æ–≤—ñ–¥–µ–π:</strong> ${correct} –∑ ${total} (${percentage}%)</p>
      <p><strong>–û—Ü—ñ–Ω–∫–∞:</strong> ${grade}</p>
      ${percentage < 70 ? '<p>–†–µ–∫–æ–º–µ–Ω–¥—É—î–º–æ –ø–æ–≤—Ç–æ—Ä–∏—Ç–∏ —Ñ–æ—Ä–º—É–ª–∏ —Ç–∞ –ø–æ–ø—Ä–∞–∫—Ç–∏–∫—É–≤–∞—Ç–∏—Å—è –≤ —Ä–æ–∑—Ä–∞—Ö—É–Ω–∫–∞—Ö</p>' : ''}
      ${percentage >= 80 ? '<p>üéâ –í–∏ –≥–æ—Ç–æ–≤—ñ –¥–æ —Å–∫–ª–∞–¥–Ω–∏—Ö –∑–∞–≤–¥–∞–Ω—å –∑ –≥—Ä–∞—Ñ—ñ–∫–æ—é!</p>' : ''}
    </div>
  `;
}

// –§—É–Ω–∫—Ü—ñ—è —Å–∫–∏–¥–∞–Ω–Ω—è —Ç–µ—Å—Ç—É
function resetQuiz() {
  userAnswers = {};
  
  // –û—á–∏—â—É—î–º–æ –≤—Å—ñ –≤–∏–±–æ—Ä–∏
  document.querySelectorAll('input[type="radio"]').forEach(input => {
    input.checked = false;
  });
  
  // –û—á–∏—â—É—î–º–æ —Å—Ç–∏–ª—ñ
  document.querySelectorAll('.answer-option').forEach(option => {
    option.classList.remove('correct', 'incorrect', 'selected');
  });
  
  // –û—á–∏—â—É—î–º–æ —Ä–µ–∑—É–ª—å—Ç–∞—Ç
  document.getElementById('quiz-result').innerHTML = '';
  
  // –°–∫–∏–¥–∞—î–º–æ –ø—Ä–æ–≥—Ä–µ—Å
  updateProgress();
}
